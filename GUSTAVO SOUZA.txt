    taxaEnergia := 0.758
    escreva("Informe o consumo do equipamento em Watts: ")
    leia(consumoWatts)

    leia(horasUso)
    escreva("Informe o número de dias de uso do equipamento: ")
    leia(diasUso)
    consumoKWh := (consumoWatts * horasUso * diasUso) / 1000

algoritmo "CalculoCorrentePotencia"
var
    resistencia, tensao, corrente, potencia: real
inicio
    escreva("Informe o valor da resistência (Ohms): ")
    leia(resistencia)
    escreva("Informe o valor da tensão (Volts): ")
    leia(tensao)
    corrente := tensao / resistencia
    potencia := tensao * corrente
    escreval("Corrente (Amperes): ", corrente)
    escreval("Resistência (Ohms): ", resistencia)
    escreval("Tensão (Volts): ", tensao)
    escreval("Potência (Watts): ", potencia)
fimalgoritmo
